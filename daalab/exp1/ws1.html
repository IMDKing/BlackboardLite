            #include "iostream"
            
            using namespace std;
            
            int binarySearch(const int array[], int low, int high, int x) {
                while (low <= high) { 
                    int mid=low + (high - low) / 2; 
                    if (array[mid]==x) 
                        return mid; 
                    if (array[mid] < x) 
                        low=mid + 1; 
                    else 
                        high=mid - 1; 
                } 
                return -1;
            } 
             
            int main() { 
                int myarray[10]; 
                int num; 
                int output; 
                cout << "Enter 10 elements in ascending order: " ; 
                for (int &i: myarray) 
                    cin>> i;
                cout << "Enter element to search: " ; 
                cin>> num;
                output = binarySearch(myarray, 0, 9, num);
                if (output == -1)
                    cout << "No Match Found" << endl;
                else 
                    cout << "Match found at position: " << output << endl; 
                return 0; 
            }
